import dns.immutable
import dns.rdtypes.euibase


@dns.immutable.immutable
class EUI48(dns.rdtypes.euibase.EUIBase):
    """EUI48 record"""

    # see: rfc7043.txt

    byte_len = 6  # 0123456789ab (in hex)
    text_len = byte_len * 3 - 1  # 01-23-45-67-89-ab
